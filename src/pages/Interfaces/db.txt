-- token blacklist table
CREATE TABLE token_blacklist (
    id SERIAL PRIMARY KEY,
    token VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- user table
CREATE TABLE users (
    id UUID PRIMARY KEY,
    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    profile_image VARCHAR(255),
    user_type INT NOT NULL,
    role VARCHAR(50),
    provider_requested BOOLEAN DEFAULT FALSE NOT NULL,
    is_active INT DEFAULT 1 NOT NULL,
    provider_status INT DEFAULT 1 NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP.
    bank_iban VARCHAR(50),
    bank_owner_name VARCHAR(100),
    is_open BOOLEAN default True
);

-- company table
CREATE TABLE company (
    id SERIAL PRIMARY KEY,
    user_id UUID,
    company_type VARCHAR(50),
    company_name VARCHAR(255),
    company_phone VARCHAR(255),
    tax_number VARCHAR(50),
    tax_office VARCHAR(255),
    address TEXT,
    tc_no VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- car table
CREATE TABLE cars (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255),
    brand VARCHAR(255),
    model VARCHAR(255),
    plate VARCHAR(50) ,
    color VARCHAR(50) ,
    year INT ,
    fuel_type VARCHAR(50),
    description TEXT,
    owner_id UUID ,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- address table
CREATE TABLE address (
    id SERIAL PRIMARY KEY,
    user_id UUID, 
    title VARCHAR(255) NOT NULL,
    street VARCHAR(255) NOT NULL,
    neighborhood VARCHAR(255) NOT NULL,
    county VARCHAR(255) NOT NULL,
    city VARCHAR(255) NOT NULL,
    description TEXT,
    distance FLOAT,
    latitude DOUBLE PRECISION,
    longitude DOUBLE PRECISION,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- work_times table
CREATE TABLE work_times (
    id SERIAL PRIMARY KEY,
    user_id UUID NOT NULL,
    start_hour TIME NOT NULL,
    end_hour TIME NOT NULL,
    days BOOLEAN[] NOT NULL CHECK (array_length(days, 1) = 7),
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- support table

CREATE TABLE support (
    id SERIAL PRIMARY KEY,
    topic VARCHAR(255),
    description VARCHAR(255),
    user_id UUID,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
)


-- services table
CREATE TABLE services (
    id SERIAL PRIMARY KEY,
    user_id UUID,
    user_services JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    updated_at TIMESTAMP
);

-- service_history
CREATE TABLE service_history (
    id SERIAL PRIMARY KEY,
    appointment_type INTEGER NOT NULL,
    customer_id UUID NOT NULL,
    company_id UUID NOT NULL,
    car_id INTEGER NOT NULL,
    address_id INTEGER,
    scheduled_time TIMESTAMP NOT NULL,
    latitude DECIMAL(9,6),
    longitude DECIMAL(9,6),
    
    service_status INTEGER NOT NULL DEFAULT 0, 
    payment_status INTEGER NOT NULL DEFAULT 0,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ratings
CREATE TABLE ratings (
    id SERIAL PRIMARY KEY,
    appointment_id INTEGER NOT NULL,
    rater_user_id UUID NOT NULL,
    target_user_id UUID NOT NULL,
    rating INTEGER NOT NULL CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
